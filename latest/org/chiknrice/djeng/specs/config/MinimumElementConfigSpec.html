    Minimum Element Config

    Project maintained by chiknriceTheme by mattgraham

    Every element (within message-elements) requires at least an 'index' and 'codec' attributes.

    Missing 'index'

    The following configuration would result in cvc-complex-type.4: Attribute 'index' must appear on element 'element'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element codec="string"/>
    </message-elements>

</config>

    Missing 'codec'

    The following configuration would result in cvc-complex-type.4: Attribute 'codec' must appear on element 'element'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1"/>
    </message-elements>

</config>
