    Minimum Codec Config

    Project maintained by chiknriceTheme by mattgraham

    Every codec (including filters) requires at least an 'id' and 'class' attributes.

    Missing 'id'

    The following configuration would result in cvc-complex-type.4: Attribute 'id' must appear on element 'element-codec'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.fin.PlainCompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
    </message-elements>

</config>

    Missing 'class'

    The following configuration would result in cvc-complex-type.4: Attribute 'class' must appear on element 'element-codec'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string"/>
        <composite-codec id="message" class="org.chiknrice.djeng.fin.PlainCompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
    </message-elements>

</config>
