    Unique Element Index

    Project maintained by chiknriceTheme by mattgraham

    A sub-element's index attribute should be unique within the message-elements as well as any composite. Both element and composite are considered a sub-element.

    Two elements under message-elements are having the same index

    The following configuration would result in Duplicate unique value [1] declared for identity constraint of element "message-elements". 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
        <element index="1" codec="string"/>
    </message-elements>

</config>

    An element and a composite under message-elements are having the same index

    The following configuration would result in Duplicate unique value [4] declared for identity constraint of element "message-elements". 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="c" class="org.chiknrice.djeng.CompositeCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
        <element index="4" codec="string"/>
        <composite index="4" codec="c">
            <element index="1" codec="string"/>
        </composite>
    </message-elements>

</config>

    Two elements under composite are having the same index

    The following configuration would result in Duplicate unique value [2] declared for identity constraint of element "composite". 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="c" class="org.chiknrice.djeng.CompositeCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
        <composite index="4" codec="c">
            <element index="2" codec="string"/>
            <element index="2" codec="string"/>
        </composite>
    </message-elements>

</config>

    An element and a composite under another composite are having the same index

    The following configuration would result in Duplicate unique value [3] declared for identity constraint of element "composite". 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="c" class="org.chiknrice.djeng.CompositeCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
        <composite index="4" codec="c">
            <element index="3" codec="string"/>
            <composite index="3" codec="c">
                <element index="6" codec="string"/>
            </composite>
        </composite>
    </message-elements>

</config>
