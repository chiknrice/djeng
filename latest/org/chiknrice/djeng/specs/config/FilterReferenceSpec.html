    Valid Filter Reference

    Project maintained by chiknriceTheme by mattgraham

    Every codec using any filters should have their code attribute reference an existing filter.

    Missing Filter

    The following configuration would result in Key 'codec-filter-id-ref' with value 'f' not found for identity constraint of element 'config'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <codec-filter id="lvar" class="org.chiknrice.djeng.fin.LengthPrefixCodecFilter"/>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec">
            <filter codec="f"/>
        </element-codec>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
    </message-elements>

</config>

    Invalid Filter

    The following configuration would result in Key 'codec-filter-id-ref' with value 'string' not found for identity constraint of element 'config'. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <codec-filter id="lvar" class="org.chiknrice.djeng.fin.LengthPrefixCodecFilter"/>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec"/>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
        <composite-codec id="some-composite" class="org.chiknrice.djeng.CompositeCodec">
            <filter codec="string"/>
        </composite-codec>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
    </message-elements>

</config>

    Valid Reference

    The following configuration would be successful. 

    <?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://www.chiknrice.org/djeng">

    <codecs>
        <codec-filter id="lvar" class="org.chiknrice.djeng.fin.LengthPrefixCodecFilter"/>
        <codec-filter id="array" class="org.chiknrice.djeng.ArrayCodecFilter"/>
        <element-codec id="string" class="org.chiknrice.djeng.fin.StringCodec">
            <filter codec="lvar"/>
        </element-codec>
        <composite-codec id="message" class="org.chiknrice.djeng.CompositeCodec"/>
        <composite-codec id="arr-lvar-composite" class="org.chiknrice.djeng.CompositeCodec">
            <filter codec="lvar"/>
            <filter codec="array"/>
        </composite-codec>
    </codecs>
    <message-elements codec="message">
        <element index="1" codec="string"/>
        <composite index="2" codec="arr-lvar-composite">
            <element index="1" codec="string"/>
        </composite>
    </message-elements>

</config>
